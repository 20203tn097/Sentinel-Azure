// Usage Instruction : 
// Paste below query in log analytics, click on Save button and select as Function from drop down by specifying function name and alias as SuperShieldActivity.
// Function usually takes 10-15 minutes to activate. You can then use function alias from any other queries (e.g. SuperShieldActivity | take 10).
// Reference : Using functions in Azure monitor log queries : https://docs.microsoft.com/azure/azure-monitor/log-query/functions
CommonSecurityLog
| where DeviceVendor == 'PCMatic'
| extend Category = extract(@'cat=(.*?);', 1, AdditionalExtensions)
| extend Severity = extract(@'Severity=(.*?);', 1, AdditionalExtensions, typeof(int))
| extend Host = extract(@'Host=(.*?);', 1, AdditionalExtensions)
| extend DeviceExternalID = extract(@'deviceExternalID=(.*?);', 1, AdditionalExtensions)
| extend GroupName = extract(@'groupName=(.*?);', 1, AdditionalExtensions)
| extend GroupID = extract(@'groupID=(.*?);', 1, AdditionalExtensions)
| extend ParentPath = extract(@'parentPath=(.*?);', 1, AdditionalExtensions)
| extend MD5 = extract(@'md5=(.*?);', 1, AdditionalExtensions)
| extend FileHash = extract(@'filehash=(.*?);', 1, AdditionalExtensions)
| extend IsUnknown = extract(@'isUnknown=(.*?);', 1, AdditionalExtensions)
| extend IsScript = extract(@'isScript=(.*?);', 1, AdditionalExtensions)
| extend UserDecided = extract(@'userDecided=(.*?);', 1, AdditionalExtensions)
| extend UserDecidedDesc = extract(@'userDecidedDesc=(.*?);', 1, AdditionalExtensions)
| extend CmdLineArgs = extract(@'cmdLineArgs=(.*?);', 1, AdditionalExtensions)
| extend Vendor = extract(@'vendor=(.*?);', 1, AdditionalExtensions)
| extend Product = extract(@'product=(.*?);', 1, AdditionalExtensions)
| extend ProductVersion = extract(@'productVersion=(.*?);', 1, AdditionalExtensions)
| extend FileCopyright = extract(@'fileCopyright=(.*?);', 1, AdditionalExtensions)
| extend IsSigned = extract(@'isSigned=(.*?);', 1, AdditionalExtensions)
| extend DigitalSigAuthority = extract(@'digitalSigAuthority=(.*?);', 1, AdditionalExtensions)
| extend DigitalSigVendor = extract(@'digitalSigVendor=(.*?);', 1, AdditionalExtensions)
| extend OS = extract(@'os=(.*?);', 1, AdditionalExtensions)
| extend OSArch = extract(@'osArch=(.*?);', 1, AdditionalExtensions)
| extend OSPlatform = extract(@'osPlatform=(.*?);', 1, AdditionalExtensions)
| extend OSRelease = extract(@'osRelease=(.*?);', 1, AdditionalExtensions)
| extend OSKernel = extract(@'osKernel=(.*?);', 1, AdditionalExtensions)
| extend ProtectionMode = extract(@'protectionMode=(.*)', 1, AdditionalExtensions)
| order by TimeGenerated desc
;
